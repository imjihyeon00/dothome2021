04. switch문 

{
    document.write("*** 04. switch문 ***");
    let num = 100;

    switch ( num ) {
        case 100:
            document.write("조건문이 실행되었습니다.(num == 100)");        
        break;

        case 200:
            document.write("조건문이 실행되었습니다.(num == 200)");        
        break;

        case 300:
            document.write("조건문이 실행되었습니다.(num == 300)");        
        break;

        default:
            document.write("조건문이 실행되었습니다.(num == 값이 없음)");        
        break;
    }
}

13. for in문 

{
    document.write("*** 13. for in문 ***
");

    //배열
    const num = [100,200,300,400,500];

    for(let i in num){
        document.write(num[i]);
    }

    //객체
    const obj = {
        a: 100,
        b: 200,
        c: 300,
        d: 400,
        e: 500
    };

    for(let key in obj){
        document.write(key + " : " + obj[key]);
    }
}

15. continue문 

{
    document.write("*** 15. continue문 ***
");

    let num = 0;
    while(true){
        num++;
        if(num == 3){
            continue;
        }
        if(num > 19){
            break;
        }
        document.write(num);
    }
}
