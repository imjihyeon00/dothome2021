01. 변수 : 데이터 저장소 

{
    var x = 100;            //변수 x에 숫자 100을 저장
    var y = 200;            //변수 y에 숫자 200을 저장
    var z = "javascript";   //변수 z에 문자 javascript를 저장

    document.write("*** 01. 변수 ***");     //도큐먼트에 출력
    document.write(x);
    document.write(y);
    document.write(z);
}

02. 변수 : 데이터 저장 + 데이터 변경 

{

03. 변수 : 데이터 저장 + 데이터 변경 + 데이터 추가 

{
    let x = 100;
    let y = 200;
    let z = "javascript";
    x += 300;         //x=x+300
    y += 400;         //y=y+400
    z += "jquery";    //z=z+"jquery"

    document.write("*** 03. 변수 ***");
    document.write(x);
    document.write(y);
    document.write(z);
}

16. 객체 : 데이터 저장(키와 값) : 표현 방법8 : 객체 속에 함수 

{
    const obj = {
        a: 100,
        b: [200, 300],
        c: "javascript",
        d: function(){
            document.write("자바스크립트가 실행 되었습니다.
");
        }
        e: function(){
            document.write(obj.c + "가 실행 되었습니다.
");
        },
        f: function(){
            document.write(this.c + "가 실행 되었습니다.
");
        }  
    }

    document.write("*** 16. 객체 ***");
    document.write(obj.a);
    document.write(obj.b);
    document.write(obj.b[0]);
    document.write(obj.b[1]);
    document.write(obj.c);
    obj.d();
    obj.e();
    obj.f();
}

01. 변수의 종류 : 지역변수 / 전역변수 

{
    document.write("*** 17. 변수의 종류 ***);

    let x = 100;               //변수 설정 : 전역 변수
    let y = 200;
    let z = "javascript"
    let a = 300;

    //선언적 함수 안에 넣고 실행
    function func() {
        let x = 100;            //변수 설정 : 지역 변수
        let y = 200;
        let z = "javascript"
        a = 400;               //전역 변수 a의 값 변경
        y = 500;               //func()안의 지역변수 y의 값 변경

        document.write("함수);
        document.write(x);
        document.write(y);
        document.write(z);
        document.write(a);    //전역변수 a를 출력
    }

    func();
    document.write();
    document.write("함수);
    document.write(x);
    document.write(y);
    document.write(z);
    document.write(a);
}

02. 변수의 종류 : 형 변환 

{
    let x = 100;
    let y = "100";

    document.write("*** 18. 변수의 변환 ***");
    document.write("x: " + x + " : " + typeof (x), ");
    document.write("y: " + y + " : " + typeof (y));

    x = "" + x; //숫자를 문자열로 형 변환
    document.write("x: " + x + " : " + typeof (x));
    y = Number(y);  //내장 객체를 이용한 형 변환
    document.write("y: " + y + " : " + typeof (y));

    x = Number(x);
    document.write("x: " + x + " : " + typeof (x));
    y = String(y);
    document.write("y: " + y + " : " + typeof (y));
}

03. 변수의 유형 : 데이터 타입 

{
    //변수 : 숫자(number), 문자(string), 함수(function), 
    //       배열(object), 객체(object), 불린(boolen(true,false)), 
    //       특수값(null, undefind), 심볼(symbol)

    document.write("*** 18. 변수의 유형 ***");
    let num1 = 100;
    let str = "javascript";
    let func = function () { };
    let arr = [1, 2, 3];
    let obj = { a: 1, b: 2, c: 3 };
    let bool = true;
    let nul;  // 아무런 값이 없을 때
    let und = undefined; //값이 정의 되어 있지 않을 때
    let sym = Symbol();

    document.write("num1: " + typeof (num1));
    document.write("str: " + typeof (str));
    document.write("func: " + typeof (func));
    document.write("arr: " + typeof (arr));
    document.write("obj: " + typeof (obj));
    document.write("bool: " + typeof (bool));
    document.write("nul: " + typeof (nul));
    document.write("und: " + typeof (und));
    document.write("sym: " + typeof (sym));
}
